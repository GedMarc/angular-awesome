FormatDate Component v1.0.1

provides format date functionality for Angular applications.

Overview

Import

Import the formatDate directive from the angular-awesome package:

// Import the formatDate directive
import { WaFormatDateDirective } from 'angular-awesome/format-date';

Examples

Format Date Examples

Basic Usage

<wa-format-date [date]="myDate"></wa-format-date>

// In your component
export class FormatDateExampleComponent {
  myDate = new Date();
}

With Two-Way Binding

<wa-format-date [(ngModel)]="selectedDate"></wa-format-date>

// In your component
export class FormatDateExampleComponent {
  selectedDate = new Date();
}

Date Formatting Options

Year, Month, Day Formatting

<!-- Full date (e.g., "January 1, 2023") -->
<wa-format-date 
  [date]="myDate"
  year="numeric"
  month="long"
  day="numeric">
</wa-format-date>

<!-- Short date (e.g., "Jan 1, 2023") -->
<wa-format-date 
  [date]="myDate"
  year="numeric"
  month="short"
  day="numeric">
</wa-format-date>

<!-- Numeric date (e.g., "1/1/2023") -->
<wa-format-date 
  [date]="myDate"
  year="numeric"
  month="numeric"
  day="numeric">
</wa-format-date>

<!-- 2-digit date (e.g., "01/01/23") -->
<wa-format-date 
  [date]="myDate"
  year="2-digit"
  month="2-digit"
  day="2-digit">
</wa-format-date>

Relative Time Example

<div>
  <p>
    Event: {{ eventName }}
  </p>
  <p>
    Date: 
    <wa-format-date 
      [date]="eventDate"
      year="numeric"
      month="long"
      day="numeric"
      hour="numeric"
      minute="numeric">
    </wa-format-date>
  </p>
  <p>
    <!-- You can combine with your own relative time logic -->
    {{ getRelativeTimeString(eventDate) }}
  </p>
</div>

// In your component
export class FormatDateExampleComponent {
  eventName = 'Annual Conference';
  eventDate = new Date('2023-12-15T09:00:00');
  
  getRelativeTimeString(date: Date): string {
    const now = new Date();
    const diffMs = date.getTime() - now.getTime();
    const diffDays = Math.floor(diffMs / (1000 * 60 * 60 * 24));
    
    if (diffDays > 0) {
      return `Starts in ${diffDays} days`;
    } else if (diffDays < 0) {
      return `Ended ${Math.abs(diffDays)} days ago`;
    } else {
      return 'Happening today!';
    }
  }
}
