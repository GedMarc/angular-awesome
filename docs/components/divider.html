Divider Component v1.0.1

provides divider functionality for Angular applications.

Overview

Import

Import the divider directive from the angular-awesome package:

// Import the divider directive
import { WaDividerDirective } from 'angular-awesome/divider';

Examples

Divider Examples

Basic Usage

<wa-divider></wa-divider>

Horizontal Divider with Custom Styling

<!-- Custom color -->
<wa-divider [color]="'#3f51b5'"></wa-divider>

<!-- Custom width -->
<wa-divider [width]="'4px'"></wa-divider>

<!-- Custom spacing -->
<wa-divider [spacing]="'2rem'"></wa-divider>

<!-- Combined custom styling -->
<wa-divider 
  [color]="'#e91e63'" 
  [width]="'3px'" 
  [spacing]="'1.5rem'">
</wa-divider>

Divider with Dynamic Styling

// In your component
export class DividerExampleComponent {
  dividerColor = '#673ab7';
  dividerWidth = '2px';
  dividerSpacing = '1rem';
  isVertical = false;
  
  toggleOrientation() {
    this.isVertical = !this.isVertical;
  }
  
  updateColor(color: string) {
    this.dividerColor = color;
  }
}

<wa-divider 
  [vertical]="isVertical"
  [color]="dividerColor"
  [width]="dividerWidth"
  [spacing]="dividerSpacing">
</wa-divider>

<div class="controls">
  <wa-button (click)="toggleOrientation()">
    Toggle Orientation
  </wa-button>
  <wa-color-picker [(ngModel)]="dividerColor"></wa-color-picker>
</div>

Divider with Responsive Styling

<wa-divider 
  [width]="isMobile ? '1px' : '3px'"
  [spacing]="isMobile ? '0.5rem' : '1.5rem'">
</wa-divider>

// In your component
export class DividerExampleComponent {
  isMobile = window.innerWidth < 768;
  
  @HostListener('window:resize')
  onResize() {
    this.isMobile = window.innerWidth < 768;
  }
}
