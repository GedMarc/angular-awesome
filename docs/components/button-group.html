ButtonGroup Component v1.0.1

provides button group functionality for Angular applications.

Overview

Import

Import the buttonGroup directive from the angular-awesome package:

// Import the buttonGroup directive
import { WaButtonGroupDirective } from 'angular-awesome/button-group';

Basic Usage

<wa-button-group label="Alignment">
  <wa-button>Left</wa-button>
  <wa-button>Center</wa-button>
  <wa-button>Right</wa-button>
</wa-button-group>

  <button waButton>Left</button>

With Angular Bindings

<wa-button-group 
  [label]="groupLabel" 
  [variant]="groupVariant" 
  [size]="groupSize" 
  [orientation]="groupOrientation">
  <wa-button *ngFor="let option of options">{{ option }}</wa-button>
</wa-button-group>

@Component({
  // ...
})
export class MyComponent {
  groupLabel = 'Dynamic Group';
  groupVariant = 'brand';
  groupSize = 'medium';
  groupOrientation = 'horizontal';
  options = ['Option 1', 'Option 2', 'Option 3'];
}

/* Add spacing between button groups */
